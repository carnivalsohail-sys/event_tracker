SETUP GOOGLE SHEETS (One-Time Only)

Follow these steps to connect your Google Sheet:

STEP 1: Create Google Sheet

1. Open Google Sheets

2. Create a new blank spreadsheet

3. Name it "Event Works Tracker"

STEP 2: Add Column Headers

In Row 1, add these headers (copy-paste):

Date | Month | Time | Location | Venue | Client | Materials | EventType | Payment | Contact | Team | Hall | Stall | FollowUp | Notes | Status | SetupStatus | ID
STEP 3: Create Google Apps Script

1. In Google Sheets, click Extensions → Apps Script

2. Delete everything in the code editor

3. Copy and paste this code:

function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Sheet1'); const data = sheet.getDataRange().getDisplayValues(); const headers = data[0]; const rows = data.slice(1); const jsonData = rows.map(row => { let obj = {}; headers.forEach((header, index) => { obj[header] = row[index]; }); return obj; }); return ContentService.createTextOutput(JSON.stringify(jsonData)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Sheet1'); const data = JSON.parse(e.postData.contents); if (data.action === 'add') { sheet.appendRow([ data.date, data.month, data.time, data.location, data.venue, data.client, data.materials, data.eventType, data.payment, data.contact, data.team, data.hall, data.stall, data.followUp, data.notes, data.status, data.setupStatus, data.id ]); return ContentService.createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } if (data.action === 'update') { const allData = sheet.getDataRange().getValues(); for (let i = 1; i < allData.length; i++) { if (allData[i][17] == data.id) { sheet.getRange(i + 1, 1, 1, 18).setValues([[ data.date, data.month, data.time, data.location, data.venue, data.client, data.materials, data.eventType, data.payment, data.contact, data.team, data.hall, data.stall, data.followUp, data.notes, data.status, data.setupStatus, data.id ]]); break; } } return ContentService.createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } if (data.action === 'delete') { const allData = sheet.getDataRange().getValues(); for (let i = 1; i < allData.length; i++) { if (allData[i][17] == data.id) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({success: false, error: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
STEP 4: Deploy the Script

1. Click Deploy → New deployment

2. Click the gear icon next to "Select type"

3. Choose Web app

4. Set "Who has access" to Anyone

5. Click Deploy

6. Copy the Web app URL that appears
